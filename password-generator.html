function makePassword() { const upper = "ABCDEFGHJKLMNPQRSTUVWXYZ"; const lower = "abcdefghijkmnpqrstuvwxyz"; const num = "23456789"; const spec = "!@#$%^&*"; // 隨機選出一個特殊符號 (確保只有一個) const randomSpec = spec[Math.floor(Math.random() * spec.length)]; // 隨機選出一個大寫字母 (確保第一個字是大寫) const firstChar = upper[Math.floor(Math.random() * upper.length)]; // 剩餘的 10 個字元 (從 upper, lower, num 中選出，不包含特殊符號) const remainingChars = upper + lower + num; let password = [firstChar]; for (let i = 0; i < 10; i++) { password.push(remainingChars[Math.floor(Math.random() * remainingChars.length)]); } // 將特殊符號隨機插入剩餘的位置 (除了第一個位置) const randomIndex = Math.floor(Math.random() * 10) + 1; password.splice(randomIndex, 0, randomSpec); document.getElementById("result").value = password.join(''); } function copyPassword() { const copyText = document.getElementById("result"); if (copyText.value) { navigator.clipboard.writeText(copyText.value).then(() => { const status = document.getElementById("status"); status.classList.add("show"); setTimeout(() => status.classList.remove("show"), 2000); }); } }
